Value at Risk (VaR) Calculation


// Monte Carlo simulation for portfolio VaR
const calculateVaR = (portfolio, confidence = 0.95, simulations = 10000) => {
  const returns = [];
  
  // Run simulations
  for (let i = 0; i < simulations; i++) {
    const simulatedReturn = runMonteCarloSimulation(portfolio);
    returns.push(simulatedReturn);
  }
  
  // Sort and find VaR at confidence level
  returns.sort((a, b) => a - b);
  const varIndex = Math.floor((1 - confidence) * simulations);
  const valueAtRisk = returns[varIndex];
  
  return {
    var: valueAtRisk,
    confidence: confidence,
    portfolioValue: portfolio.totalValue,
    maxLoss: portfolio.totalValue * Math.abs(valueAtRisk)
  };
};
